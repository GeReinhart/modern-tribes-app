### Background Task Service (`backend/services/batch_email_service.py`)
```python
import asyncio
from motor.motor_asyncio import AsyncIOMotorClient
from typing import List, Dict
import logging
from datetime import datetime

logger = logging.getLogger(__name__)

class BatchEmailService:
    def __init__(self, db: AsyncIOMotorClient):
        self.db = db

    async def send_batch_emails(
        self,
        entity_collection: str,
        query: Dict,
        subject: str,
        template: str,
        batch_id: str,
        batch_size: int = 50  # Send in chunks
    ):
        """Send emails in batches with progress tracking"""

        # Update batch status to processing
        await self.db.email_batches.update_one(
            {"_id": batch_id},
            {
                "$set": {
                    "status": "processing",
                    "started_at": datetime.utcnow()
                }
            }
        )

        total_sent = 0
        total_failed = 0

        try:
            # Get entities with email field
            cursor = self.db[entity_collection].find(query)
            entities = await cursor.to_list(length=None)
            total = len(entities)

            # Process in batches
            for i in range(0, len(entities), batch_size):
                batch = entities[i:i + batch_size]

                # Send emails concurrently within batch
                tasks = []
                for entity in batch:
                    if email := entity.get("email"):
                        # Personalize email body
                        body = template.format(
                            name=entity.get("name", "User"),
                            email=email,
                            **entity  # Allow any field from entity
                        )
                        tasks.append(
                            self._send_with_retry(email, subject, body)
                        )

                # Wait for batch to complete
                results = await asyncio.gather(*tasks, return_exceptions=True)

                # Count results
                for result in results:
                    if isinstance(result, Exception):
                        total_failed += 1
                        logger.error(f"Email failed: {result}")
                    else:
                        total_sent += 1

                # Update progress
                progress = ((i + len(batch)) / total) * 100
                await self.db.email_batches.update_one(
                    {"_id": batch_id},
                    {
                        "$set": {
                            "progress": progress,
                            "sent": total_sent,
                            "failed": total_failed
                        }
                    }
                )

                # Rate limiting - wait between batches
                await asyncio.sleep(1)

            # Mark as completed
            await self.db.email_batches.update_one(
                {"_id": batch_id},
                {
                    "$set": {
                        "status": "completed",
                        "completed_at": datetime.utcnow(),
                        "total": total,
                        "sent": total_sent,
                        "failed": total_failed
                    }
                }
            )

        except Exception as e:
            logger.error(f"Batch email failed: {e}")
            await self.db.email_batches.update_one(
                {"_id": batch_id},
                {
                    "$set": {
                        "status": "failed",
                        "error": str(e),
                        "completed_at": datetime.utcnow()
                    }
                }
            )

    async def _send_with_retry(
        self,
        email: str,
        subject: str,
        body: str,
        max_retries: int = 3
    ):
        """Send email with retry logic"""
        from services.email_service import send_email

        for attempt in range(max_retries):
            try:
                await send_email(email, subject, body)
                return True
            except Exception as e:
                if attempt == max_retries - 1:
                    raise
                await asyncio.sleep(2 ** attempt)  # Exponential backoff
```

### API Routes (`backend/routes/batch_email.py`)
```python
from fastapi import APIRouter, BackgroundTasks, Depends, HTTPException
from pydantic import BaseModel, EmailStr
from typing import Optional, Dict, List
from datetime import datetime
import uuid

router = APIRouter(prefix="/api/batch-emails", tags=["batch-emails"])

class BatchEmailRequest(BaseModel):
    collection: str  # Entity collection name
    query: Optional[Dict] = {}  # MongoDB query filter
    subject: str
    template: str  # Email template with placeholders: {name}, {email}, etc.

class BatchEmailResponse(BaseModel):
    batch_id: str
    status: str
    message: str

@router.post("/send", response_model=BatchEmailResponse)
async def send_batch_emails(
    request: BatchEmailRequest,
    background_tasks: BackgroundTasks,
    db = Depends(get_database)
):
    """Trigger batch email sending"""

    # Validate collection exists
    collections = await db.list_collection_names()
    if request.collection not in collections:
        raise HTTPException(404, "Collection not found")

    # Create batch record
    batch_id = str(uuid.uuid4())
    batch_record = {
        "_id": batch_id,
        "collection": request.collection,
        "query": request.query,
        "subject": request.subject,
        "status": "queued",
        "progress": 0,
        "sent": 0,
        "failed": 0,
        "created_at": datetime.utcnow()
    }

    await db.email_batches.insert_one(batch_record)

    # Add to background tasks
    service = BatchEmailService(db)
    background_tasks.add_task(
        service.send_batch_emails,
        request.collection,
        request.query,
        request.subject,
        request.template,
        batch_id
    )

    return BatchEmailResponse(
        batch_id=batch_id,
        status="queued",
        message="Batch email job queued successfully"
    )

@router.get("/status/{batch_id}")
async def get_batch_status(
    batch_id: str,
    db = Depends(get_database)
):
    """Get batch email status"""
    batch = await db.email_batches.find_one({"_id": batch_id})

    if not batch:
        raise HTTPException(404, "Batch not found")

    return batch

@router.get("/history")
async def get_batch_history(
    skip: int = 0,
    limit: int = 20,
    db = Depends(get_database)
):
    """Get batch email history"""
    cursor = db.email_batches.find().sort("created_at", -1).skip(skip).limit(limit)
    batches = await cursor.to_list(length=limit)
    return batches
```

## 2. Frontend Implementation

### Batch Email Component (`frontend/src/components/BatchEmail.jsx`)
```jsx
import { useState, useEffect } from 'react';
import { Send, Clock, CheckCircle, XCircle } from 'lucide-react';

export default function BatchEmail() {
  const [formData, setFormData] = useState({
    collection: 'users',
    query: '{}',
    subject: '',
    template: 'Hello {name},\n\nThis is a test email sent to {email}.',
  });

  const [batches, setBatches] = useState([]);
  const [activeBatch, setActiveBatch] = useState(null);
  const [loading, setLoading] = useState(false);

  // Submit batch email
  const handleSubmit = async (e) => {
    e.preventDefault();
    setLoading(true);

    try {
      const response = await fetch('/api/batch-emails/send', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          ...formData,
          query: JSON.parse(formData.query),
        }),
      });

      const data = await response.json();

      if (response.ok) {
        setActiveBatch(data.batch_id);
        alert('Batch email started!');
        fetchBatchHistory();
      }
    } catch (error) {
      alert('Error: ' + error.message);
    } finally {
      setLoading(false);
    }
  };

  // Fetch batch history
  const fetchBatchHistory = async () => {
    const response = await fetch('/api/batch-emails/history');
    const data = await response.json();
    setBatches(data);
  };

  // Poll active batch status
  useEffect(() => {
    if (!activeBatch) return;

    const interval = setInterval(async () => {
      const response = await fetch(`/api/batch-emails/status/${activeBatch}`);
      const data = await response.json();

      if (data.status === 'completed' || data.status === 'failed') {
        setActiveBatch(null);
        fetchBatchHistory();
      }
    }, 2000);

    return () => clearInterval(interval);
  }, [activeBatch]);

  useEffect(() => {
    fetchBatchHistory();
  }, []);

  return (
    <div className="max-w-4xl mx-auto p-6 space-y-6">
      {/* Form */}
      <div className="bg-white rounded-lg shadow p-6">
        <h2 className="text-2xl font-bold mb-4">Send Batch Emails</h2>

        <form onSubmit={handleSubmit} className="space-y-4">
          <div>
            <label className="block text-sm font-medium mb-1">
              Collection Name
            </label>
            <input
              type="text"
              value={formData.collection}
              onChange={(e) => setFormData({...formData, collection: e.target.value})}
              className="w-full border rounded-lg px-3 py-2"
              placeholder="users"
            />
          </div>

          <div>
            <label className="block text-sm font-medium mb-1">
              Query Filter (MongoDB JSON)
            </label>
            <textarea
              value={formData.query}
              onChange={(e) => setFormData({...formData, query: e.target.value})}
              className="w-full border rounded-lg px-3 py-2 font-mono text-sm"
              rows="3"
              placeholder='{"status": "active"}'
            />
          </div>

          <div>
            <label className="block text-sm font-medium mb-1">
              Subject
            </label>
            <input
              type="text"
              value={formData.subject}
              onChange={(e) => setFormData({...formData, subject: e.target.value})}
              className="w-full border rounded-lg px-3 py-2"
              required
            />
          </div>

          <div>
            <label className="block text-sm font-medium mb-1">
              Email Template (use {'{'}name{'}'}, {'{'}email{'}'}, etc.)
            </label>
            <textarea
              value={formData.template}
              onChange={(e) => setFormData({...formData, template: e.target.value})}
              className="w-full border rounded-lg px-3 py-2"
              rows="6"
              required
            />
          </div>

          <button
            type="submit"
            disabled={loading}
            className="w-full bg-blue-600 text-white rounded-lg px-4 py-2 flex items-center justify-center gap-2 hover:bg-blue-700 disabled:opacity-50"
          >
            <Send size={18} />
            {loading ? 'Sending...' : 'Send Batch Emails'}
          </button>
        </form>
      </div>

      {/* History */}
      <div className="bg-white rounded-lg shadow p-6">
        <h3 className="text-xl font-bold mb-4">Batch History</h3>

        <div className="space-y-3">
          {batches.map((batch) => (
            <BatchStatus key={batch._id} batch={batch} />
          ))}
        </div>
      </div>
    </div>
  );
}

function BatchStatus({ batch }) {
  const statusIcons = {
    queued: <Clock className="text-gray-500" />,
    processing: <Clock className="text-blue-500 animate-spin" />,
    completed: <CheckCircle className="text-green-500" />,
    failed: <XCircle className="text-red-500" />,
  };

  return (
    <div className="border rounded-lg p-4">
      <div className="flex items-center justify-between">
        <div className="flex items-center gap-3">
          {statusIcons[batch.status]}
          <div>
            <div className="font-medium">{batch.subject}</div>
            <div className="text-sm text-gray-500">
              {new Date(batch.created_at).toLocaleString()}
            </div>
          </div>
        </div>

        <div className="text-right">
          {batch.status === 'processing' && (
            <div className="text-sm">
              <div className="font-medium">{batch.progress.toFixed(0)}%</div>
              <div className="text-gray-500">
                {batch.sent} sent, {batch.failed} failed
              </div>
            </div>
          )}

          {batch.status === 'completed' && (
            <div className="text-sm text-green-600 font-medium">
              ✓ {batch.sent} sent
            </div>
          )}

          {batch.status === 'failed' && (
            <div className="text-sm text-red-600 font-medium">
              ✗ Failed
            </div>
          )}
        </div>
      </div>

      {batch.status === 'processing' && (
        <div className="mt-2 bg-gray-200 rounded-full h-2">
          <div
            className="bg-blue-600 h-2 rounded-full transition-all"
            style={{ width: `${batch.progress}%` }}
          />
        </div>
      )}
    </div>
  );
}
```

## 3. Alternative: Using Celery (Production Recommended)

For larger scale, use Celery with Redis:

```python
# backend/celery_app.py
from celery import Celery

celery_app = Celery(
    "tasks",
    broker=settings.REDIS_URL,
    backend=settings.REDIS_URL
)

@celery_app.task
def send_batch_emails_task(collection, query, subject, template, batch_id):
    # Use sync version or run async in sync context
    import asyncio
    asyncio.run(
        BatchEmailService(db).send_batch_emails(
            collection, query, subject, template, batch_id
        )
    )
```

Then run Celery worker:
```bash
celery -A celery_app worker --loglevel=info
```
