## 3. Configuration

### `src/i18n/config.ts`

```typescript
import i18n from 'i18next';
import { initReactI18next } from 'react-i18next';
import LanguageDetector from 'i18next-browser-languagedetector';

// Import translations
import en from './locales/en.json';
import es from './locales/es.json';
import fr from './locales/fr.json';

const resources = {
  en: { translation: en },
  es: { translation: es },
  fr: { translation: fr },
};

i18n
  .use(LanguageDetector) // Detect user language
  .use(initReactI18next) // Pass i18n to react-i18next
  .init({
    resources,
    fallbackLng: 'en',
    debug: import.meta.env.DEV,

    interpolation: {
      escapeValue: false, // React already escapes
    },

    detection: {
      // Order of language detection
      order: ['localStorage', 'navigator'],
      caches: ['localStorage'],
    },
  });

export default i18n;
```

## 5. Usage Examples

### Basic Component

```typescript
import { useTranslation } from 'react-i18next';

export default function Home() {
  const { t, i18n } = useTranslation();

  const changeLanguage = (lng: string) => {
    i18n.changeLanguage(lng);
  };

  return (
    <div>
      <h1>{t('app.title')}</h1>
      <p>{t('app.welcome')}</p>
      <p>{t('greeting', { name: 'John' })}</p>

      <div className="flex gap-2">
        <button onClick={() => changeLanguage('en')}>EN</button>
        <button onClick={() => changeLanguage('es')}>ES</button>
        <button onClick={() => changeLanguage('fr')}>FR</button>
      </div>

      <p>Current language: {i18n.language}</p>
    </div>
  );
}
```

### Language Switcher Component

```typescript
import { useTranslation } from 'react-i18next';

const languages = {
  en: { nativeName: 'English' },
  es: { nativeName: 'Español' },
  fr: { nativeName: 'Français' },
};

export default function LanguageSwitcher() {
  const { i18n } = useTranslation();

  return (
    <select
      value={i18n.language}
      onChange={(e) => i18n.changeLanguage(e.target.value)}
      className="px-3 py-2 border rounded-lg"
    >
      {Object.entries(languages).map(([code, { nativeName }]) => (
        <option key={code} value={code}>
          {nativeName}
        </option>
      ))}
    </select>
  );
}
```

## 6. Translation Management Scripts

### `src/i18n/scripts/check-translations.ts`

```typescript
import * as fs from 'fs';
import * as path from 'path';

interface TranslationObject {
  [key: string]: string | TranslationObject;
}

const LOCALES_DIR = path.join(__dirname, '../locales');
const BASE_LANG = 'en';

function flattenObject(
  obj: TranslationObject,
  prefix = ''
): Record<string, string> {
  return Object.keys(obj).reduce((acc: Record<string, string>, key) => {
    const value = obj[key];
    const newKey = prefix ? `${prefix}.${key}` : key;

    if (typeof value === 'object' && value !== null) {
      Object.assign(acc, flattenObject(value, newKey));
    } else {
      acc[newKey] = value as string;
    }

    return acc;
  }, {});
}

function getTranslations(lang: string): TranslationObject {
  const filePath = path.join(LOCALES_DIR, `${lang}.json`);
  if (!fs.existsSync(filePath)) {
    return {};
  }
  return JSON.parse(fs.readFileSync(filePath, 'utf-8'));
}

function checkMissingTranslations() {
  const baseTranslations = getTranslations(BASE_LANG);
  const flatBase = flattenObject(baseTranslations);

  const files = fs.readdirSync(LOCALES_DIR);
  const languages = files
    .filter((file) => file.endsWith('.json') && file !== `${BASE_LANG}.json`)
    .map((file) => file.replace('.json', ''));

  let hasMissing = false;

  languages.forEach((lang) => {
    const translations = getTranslations(lang);
    const flatTranslations = flattenObject(translations);

    const missingKeys = Object.keys(flatBase).filter(
      (key) => !(key in flatTranslations)
    );

    const extraKeys = Object.keys(flatTranslations).filter(
      (key) => !(key in flatBase)
    );

    if (missingKeys.length > 0) {
      hasMissing = true;
      console.log(`\n❌ Missing translations in ${lang}:`);
      missingKeys.forEach((key) => {
        console.log(`   - ${key}: "${flatBase[key]}"`);
      });
    }

    if (extraKeys.length > 0) {
      console.log(`\n⚠️  Extra translations in ${lang} (not in base):`);
      extraKeys.forEach((key) => {
        console.log(`   - ${key}`);
      });
    }

    if (missingKeys.length === 0 && extraKeys.length === 0) {
      console.log(`\n✅ ${lang}: All translations up to date`);
    }
  });

  if (hasMissing) {
    process.exit(1);
  }
}

checkMissingTranslations();
```

### `src/i18n/scripts/generate-translations.ts`

```typescript
import * as fs from 'fs';
import * as path from 'path';

interface TranslationObject {
  [key: string]: string | TranslationObject;
}

const LOCALES_DIR = path.join(__dirname, '../locales');
const BASE_LANG = 'en';

function flattenObject(
  obj: TranslationObject,
  prefix = ''
): Record<string, string> {
  return Object.keys(obj).reduce((acc: Record<string, string>, key) => {
    const value = obj[key];
    const newKey = prefix ? `${prefix}.${key}` : key;

    if (typeof value === 'object' && value !== null) {
      Object.assign(acc, flattenObject(value, newKey));
    } else {
      acc[newKey] = value as string;
    }

    return acc;
  }, {});
}

function unflattenObject(flat: Record<string, string>): TranslationObject {
  const result: TranslationObject = {};

  Object.keys(flat).forEach((key) => {
    const keys = key.split('.');
    let current: any = result;

    keys.forEach((k, index) => {
      if (index === keys.length - 1) {
        current[k] = flat[key];
      } else {
        current[k] = current[k] || {};
        current = current[k];
      }
    });
  });

  return result;
}

function getTranslations(lang: string): TranslationObject {
  const filePath = path.join(LOCALES_DIR, `${lang}.json`);
  if (!fs.existsSync(filePath)) {
    return {};
  }
  return JSON.parse(fs.readFileSync(filePath, 'utf-8'));
}

function saveTranslations(lang: string, translations: TranslationObject) {
  const filePath = path.join(LOCALES_DIR, `${lang}.json`);
  fs.writeFileSync(filePath, JSON.stringify(translations, null, 2) + '\n');
}

function generateMissingTranslations() {
  const baseTranslations = getTranslations(BASE_LANG);
  const flatBase = flattenObject(baseTranslations);

  const files = fs.readdirSync(LOCALES_DIR);
  const languages = files
    .filter((file) => file.endsWith('.json') && file !== `${BASE_LANG}.json`)
    .map((file) => file.replace('.json', ''));

  languages.forEach((lang) => {
    const translations = getTranslations(lang);
    const flatTranslations = flattenObject(translations);

    const missingKeys = Object.keys(flatBase).filter(
      (key) => !(key in flatTranslations)
    );

    if (missingKeys.length > 0) {
      console.log(`\n📝 Adding ${missingKeys.length} missing keys to ${lang}:`);

      missingKeys.forEach((key) => {
        // Add with placeholder - you can replace with API call to translation service
        flatTranslations[key] = `[${lang.toUpperCase()}] ${flatBase[key]}`;
        console.log(`   + ${key}`);
      });

      const unflattened = unflattenObject(flatTranslations);
      saveTranslations(lang, unflattened);

      console.log(`✅ Updated ${lang}.json`);
    } else {
      console.log(`\n✅ ${lang}: No missing translations`);
    }
  });
}

generateMissingTranslations();
```

### With AI Translation (OpenAI)

```typescript
// src/i18n/scripts/generate-translations-ai.ts
import * as fs from 'fs';
import * as path from 'path';
import OpenAI from 'openai';

const openai = new OpenAI({
  apiKey: process.env.OPENAI_API_KEY,
});

const LOCALES_DIR = path.join(__dirname, '../locales');
const BASE_LANG = 'en';

const LANGUAGE_NAMES: Record<string, string> = {
  es: 'Spanish',
  fr: 'French',
  de: 'German',
  it: 'Italian',
  pt: 'Portuguese',
  ja: 'Japanese',
  zh: 'Chinese',
};

async function translateText(text: string, targetLang: string): Promise<string> {
  const response = await openai.chat.completions.create({
    model: 'gpt-4',
    messages: [
      {
        role: 'system',
        content: `You are a professional translator. Translate the following text to ${LANGUAGE_NAMES[targetLang] || targetLang}. Only return the translation, nothing else. Preserve any placeholders like {{variable}}.`,
      },
      {
        role: 'user',
        content: text,
      },
    ],
    temperature: 0.3,
  });

  return response.choices[0]?.message?.content?.trim() || text;
}

// ... (rest of the helper functions from above)

async function generateMissingTranslationsAI() {
  const baseTranslations = getTranslations(BASE_LANG);
  const flatBase = flattenObject(baseTranslations);

  const files = fs.readdirSync(LOCALES_DIR);
  const languages = files
    .filter((file) => file.endsWith('.json') && file !== `${BASE_LANG}.json`)
    .map((file) => file.replace('.json', ''));

  for (const lang of languages) {
    const translations = getTranslations(lang);
    const flatTranslations = flattenObject(translations);

    const missingKeys = Object.keys(flatBase).filter(
      (key) => !(key in flatTranslations)
    );

    if (missingKeys.length > 0) {
      console.log(`\n🤖 Translating ${missingKeys.length} keys to ${lang}...`);

      for (const key of missingKeys) {
        const originalText = flatBase[key];
        console.log(`   Translating: ${key}`);

        const translated = await translateText(originalText, lang);
        flatTranslations[key] = translated;

        // Add delay to avoid rate limits
        await new Promise(resolve => setTimeout(resolve, 200));
      }

      const unflattened = unflattenObject(flatTranslations);
      saveTranslations(lang, unflattened);

      console.log(`✅ Updated ${lang}.json`);
    }
  }
}

generateMissingTranslationsAI();
```

## 7. Package.json Scripts

```json
{
  "scripts": {
    "dev": "vite",
    "build": "tsc && vite build",
    "i18n:check": "tsx src/i18n/scripts/check-translations.ts",
    "i18n:generate": "tsx src/i18n/scripts/generate-translations.ts",
    "i18n:generate:ai": "tsx src/i18n/scripts/generate-translations-ai.ts",
    "i18n:extract": "i18next-scanner"
  }
}
```

Install `tsx` for running TypeScript scripts:

```bash
npm install -D tsx
```

## 8. Usage

### Check for missing translations:
```bash
npm run i18n:check
```

### Generate placeholders for missing translations:
```bash
npm run i18n:generate
```

### Generate AI translations (with OpenAI):
```bash
OPENAI_API_KEY=your-key npm run i18n:generate:ai
```

## 9. Optional: Auto-extract translation keys

Install scanner:
```bash
npm install -D i18next-scanner
```

### `i18next-scanner.config.js`

```javascript
module.exports = {
  input: ['src/**/*.{ts,tsx}'],
  output: './',
  options: {
    debug: false,
    func: {
      list: ['t', 'i18n.t'],
      extensions: ['.ts', '.tsx'],
    },
    lngs: ['en', 'es', 'fr'],
    defaultLng: 'en',
    defaultValue: '',
    resource: {
      loadPath: 'src/i18n/locales/{{lng}}.json',
      savePath: 'src/i18n/locales/{{lng}}.json',
      jsonIndent: 2,
    },
  },
};
```

Run extraction:
```bash
npm run i18n:extract
```
